# Plotting decison boundary

from mpl_toolkits.mplot3d import Axes3D

X_plot = scaler.inverse_transform(X_train)

x_min, x_max = X_plot[:, 0].min() - 1, X_plot[:, 0].max() + 1
y_min, y_max = X_plot[:, 1].min() - 1, X_plot[:, 1].max() + 1
z_min, z_max = X_plot[:, 2].min() - 1, X_plot[:, 2].max() + 1

xx, yy, zz = np.meshgrid(
    np.linspace(x_min, x_max, 25),
    np.linspace(y_min, y_max, 25),
    np.linspace(z_min, z_max, 25)
)

grid_points = np.c_[xx.ravel(), yy.ravel(), zz.ravel()]
grid_scaled = scaler.transform(grid_points)

Z = G3DO_M2.predict(grid_scaled)

fig = plt.figure(figsize=(8,6))
ax = fig.add_subplot(111, projection='3d')

ax.scatter(
    grid_points[:, 0],
    grid_points[:, 1],
    grid_points[:, 2],
    c=Z,
    alpha=0.05
)

ax.scatter(
    X_plot[:, 0],
    X_plot[:, 1],
    X_plot[:, 2],
    c=y_train,
    edgecolor='k'
)

ax.set_title("Decision Boundary - Gaussian 3D Overlap")
ax.set_xlabel("Feature 1")
ax.set_ylabel("Feature 2")
ax.set_zlabel("Feature 3")

plt.show()
